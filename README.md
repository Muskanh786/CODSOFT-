n1=int(input("Enter the first number"))
n2=int(input("Enter the second number"))
op=input("Choose the operator(+,-,*,/,%)")
if (op == '+'):
    print(n1+n2)
elif (op == '-'):
    print(n1-n2)
elif (op == '*'):
    print(n1*n2)
elif (op == '/'):
    print(n1/n2)
elif (op == '%') :
    print(n1%n2)
else :
    print("invalid")
